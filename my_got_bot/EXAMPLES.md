```python
from memory import ChatMemory
from brain import BrainText
from engine import think_one_step
from mouth import extract_final_answer

# –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
memory = ChatMemory()
brain = BrainText()

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –æ—Ç–≤–µ—Ç–∞
def get_bot_response(user_message):
    brain.clear()
    history = memory.get_formatted_history()
    
    for i in range(4):
        response = think_one_step(
            user_message=user_message,
            history=history,
            current_cot=brain.get_chain(),
            is_first_step=(i == 0)
        )
        brain.add_step(response)
        
        final, is_final = extract_final_answer(response)
        if is_final:
            memory.add_exchange(user_message, final)
            return final
    
    # –ï—Å–ª–∏ –Ω–µ –Ω–∞—à–ª–∏ FINAL_ANSWER
    fallback = brain.get_chain().split("\n\n")[-1]
    memory.add_exchange(user_message, fallback)
    return fallback

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
answer = get_bot_response("–ü—Ä–∏–≤–µ—Ç!")
print(answer)
```
